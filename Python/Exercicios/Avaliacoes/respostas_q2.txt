import string

def Proprio(texto):
    L=[]
    N=[]
    for pa in texto.split():
        if pa.istitle() is True:
            L.append(pa)
            n = texto.count(pa)
            N.append(n)
            
    for item in L:
        a = ""
        posicao = texto.find(item)
        while a == "":
            posicao = posicao - 1
            a = texto[posicao]
        if a in '!?.':
            b = L.index(item)
            L.pop(b)
            
    return {x:y for x,y in zip(L,N)}
   


Identificar as 5 palavras mais comuns: 

def comum(texto):
    quantos_cada = {x:texto.count(x) for x in texto.split()}
    organizado = [[x,y] for x,y in zip(quantos_cada.values(),quantos_cada.keys())]
    organizado.sort()
    decrescente = organizado[-1::-1]
    
    print('As 5 palavras mais comuns são:')
    for pala in range(5):
        print(decrescente[pala][1])
